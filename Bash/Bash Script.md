# Просто bash script. Example

```bash
echo hello

cat package.json # Выводит содержимое файла в консоль

echo hello > some.txt # Записывает в файл some.txt строку hello
rm some.txt # Удаляет файл some.txt

echo hello >> some.txt # Добавляет в файл some.txt строку hello
cat some.txt # Выводит содержимое файла some.txt в консоль

chmod u+x some.sh # Делаем файл исполняемым
```
